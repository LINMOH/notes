## 题解

题目：[P4343 [SHOI2015] 自动刷题机](https://www.luogu.com.cn/problem/P4343)

这道题看了好久才发现是二分。

注意这道题数据范围很大，容易不开 `long long` 见祖宗。最大值、最小值各跑一次二分。

```cpp
#include <cstdio>
#include <algorithm>

const int MAXN = 1e5 + 10;
int L, K;
int ls[MAXN];

long long check(long long n) {
    if (n < 1) {
        return 2e18;
    }

    long long sum = 0, cnt = 0;
    for (int i = 0; i < L; i++) {
        sum += ls[i];

        if (sum < 0) {
            sum = 0;
        }

        if (sum >= n) {
            cnt++;
            sum = 0;
        }
    }
    return cnt;
}

int main() {
    long long sum = 0;
    long long max_val = 0;

    scanf("%d%d", &L, &K);

    for (int i = 0; i < L; i++) {
        scanf("%d", &ls[i]);
        sum += ls[i];
        if (sum < 0) {
            sum = 0;
        }
        max_val = std::max(sum, max_val);
    }

    long long l = 0, r = max_val, mid = 0, ans1 = 0;

    while (l <= r) {
        mid = l + (r - l) / 2;
        if (check(mid) <= K) {
            ans1 = mid;
            r = mid - 1;
        } else {
            l = mid + 1;
        }
    }

    l = 0, r = max_val, mid = 0;
    long long ans2 = 0;

    while (l <= r) {
        mid = l + (r - l) / 2;
        if (check(mid) >= K) {
            ans2 = mid;
            l = mid + 1;
        } else {
            r = mid - 1;
        }
    }

    if (check(ans1) != K) {
        printf("-1\n");
    } else {
        printf("%lld %lld\n", ans1, ans2);
    }

    return 0;
}
```

有点冗长，但是能跑。时间复杂度是 $O(L \log \max)$。

实际上完全不用求 `max_val` 作为上界。比如 `r` 设为 `1e15` 也是能过的。
